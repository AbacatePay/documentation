## Quick Usage

```go

	"context"
	"github.com/AbacatePay/abacatepay-go-sdk/abacatepay"
	"time"
)

	// Initialize the SDK with your API key
	client, err := abacatepay.New(&abacatepay.ClientConfig{
		ApiKey:  "your_api_key",
		Timeout: 10 * time.Second,
	})
	if err != nil {
		panic(err)
	}
	
	// Use the client...
}
```

## Criando uma Cobrança

```go
package main

import (
	"context"
	"github.com/AbacatePay/abacatepay-go-sdk/abacatepay"
	"github.com/AbacatePay/abacatepay-go-sdk/v1/billing"
	"log"
	"time"
)

func main() {
	client, err := abacatepay.New(&abacatepay.ClientConfig{
		ApiKey:  "your_api_key",
		Timeout: 10 * time.Second,
	})
	if err != nil {
		panic(err)
	}

	// Criando uma Cobrança Única
	body := &billing.CreateBillingBody{
		Frequency:     billing.OneTime,
		Methods:       []billing.Method{billing.PIX},
		CompletionUrl: "https://yoursite.com/payment/success",
		ReturnUrl:     "https://yoursite.com/app",
		Products: []*billing.BillingProduct{
			{
				ExternalId:  "PRO-PLAN",
				Name:        "Pro plan",
				Description: "Pro plan subscription",
				Quantity:    1,
				Price:       1000, // Preço em centavos
			},
		},
		Customer: &billing.BillingCustomer{
			Email: "customer@example.com",
		},
	}

	ctx := context.Background()
	createResponse, err := client.Billing.Create(ctx, body)
	if err != nil {
		panic(err)
	}

	log.Println(createResponse.Data.Url) // URL de pagamento para seu cliente
}
```
